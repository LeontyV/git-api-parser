# Программа для сбора информации по участникам репозиториев на github
# версия 0.1

# пример запроса: "python main.py -url https://github.com/ytdl-org/youtube-dl -since 2020-12-12T00:00:00Z -until 2020-12-15T00:00:00Z"

# параметры:
	-url - передаем адрес репозитория на github
	-since - передаем начальную дату в формате 2020-12-12T00:00:00Z (по-умолчанию будет использоваться дата создания репозитория)
	-until - передаем конечную дату в формате 2020-12-12T00:00:00Z (по-умолчанию будет использоваться текущая дата)


# создаем класс объекта
git = GetFromGitApi(use_session=True)

# выводит на экран информацию о пользователях, которые делали коммиты в запрашиваемый репозиторий на данном промежутке времени
git.get_top_authors(sys.argv)

# выводит информацию об отрытых и закрытых пул-реквестах на данном промежутке времени
git.get_pull_requests(sys.argv)

# выводит информацию об устаревших пул-реквестах на данном промежутке времени
git.get_older_pull_requests(sys.argv)

# выводит информацию об отрытых и закрытых проблемах на данном промежутке времени
git.get_issues(sys.argv)

# выводит информацию об устаревших проблемах на данном промежутке времени
git.get_old_issues(sys.argv)
